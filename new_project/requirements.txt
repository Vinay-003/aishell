mkdir new_project
cd new_project
mkdir app
cd app

echo "flask" >> requirements.txt

cat << EOF >> app/__init__.py
from flask import Flask
app = Flask(__name__)

@app.route('/')
def home():
    return 'Hello World'

if __name__ == '__main__':
    app.run(debug=True)
EOF

echo "
    from app import app
    app.run(debug=True)

if __name__ == '__main__':
    app.run(debug=True)
" > app/main.py

pip freeze > requirements.txt
echo "# new project" > README.md
